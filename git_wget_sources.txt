# git_wget_sources.txt
# Formát záznamů:
#
#   type=git
#   url=<repo URL>
#   dir=<cílový adresář (relativní od BASE_DIR)>
#   [branch=<větev, default: výchozí větev>]
#   [depth=<--depth N, default: plný klon>]
#
#   type=wget
#   url=<URL nebo http://host/path/ pro rekurzivní stahování>
#   dir=<cílový adresář>
#   [args=<dodatečné argumenty pro wget>]
#   [accept=<přípony souborů oddělené čárkou, default: vše>]
#
# Prázdný řádek nebo řádek začínající # ukončuje záznam.
