# git_wget_sources.txt
# Formát záznamů:
#
#   type=git
#   url=<repo URL>
#   dir=<cílový adresář (relativní od BASE_DIR)>
#   [branch=<větev, default: výchozí větev>]
#   [depth=<--depth N, default: plný klon>]
#
#   type=wget
#   url=<URL>
#   dir=<cílový adresář>
#   [accept=<přípony souborů oddělené čárkou>]
#   [args=<přepínače bez hodnot s mezerami: --recursive --no-parent …>]
#   [user_agent=<celý UA string – předá se jako --user-agent=…>]
#
# Prázdný řádek nebo řádek začínající # ukončuje záznam.
